ALGORITHM distinctElement

VAR 
 COUNT : INTEGER := 0;
 OBJECT_1: HASH_TABLE<INTEGER,BOOLEAN>;
 OBJECT_2: HASH_TABLE<INTEGER,BOOLEAN>;
 TWICE: HASH_TABLE<INTEGER,BOOLEAN>;

BEGIN

 OBJECT_1: HASH_TABLE<INTEGER,BOOLEAN>;

  OBJECT_1.insert(3,true);
  OBJECT_1.insert(1,true);
  OBJECT_1.insert(7,true);
  OBJECT_1.insert(9,true);

OBJECT_2: HASH_TABLE<INTEGER,BOOLEAN>;

  OBJECT_2.insert(2,true);
  OBJECT_2.insert(4,true);
  OBJECT_2.insert(1,true);
  OBJECT_2.insert(9,true);
  OBJECT_2.insert(3,true);

FOR KEY IN OBJECT_1:
   IF (OBJECT_2[KEY]) THEN
   TWICE[KEY] = TRUE
   END IF 
END FOR 

FOR KEY IN TWICE:
   COUNT = COUNT + KEY
END_FOR
   Write(COUNT)


END
